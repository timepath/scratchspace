00026 bool Nagger_SendEntity(entity this, entity to, float sendflags)
00027 {
00028     int nags, i, f, b;
00029     entity e;
00030     WriteHeader(MSG_ENTITY, ENT_CLIENT_NAGGER);
00031 
00032     // bits:
00033     //   1 = ready
00034     //   2 = player needs to ready up
00035     //   4 = vote
00036     //   8 = player needs to vote
00037     //  16 = warmup
00038     // sendflags:
00039     //  64 = vote counts
00040     // 128 = vote string
00041 
00042     nags = 0;
00043     if (readycount)
00044     {
00045         nags |= BIT(0);
00046         if (to.ready == 0) nags |= BIT(1);
00047     }
00048     if (vote_called)
00049     {
00050         nags |= BIT(2);
00051         if (to.vote_selection == 0) nags |= BIT(3);
00052     }
00053     if (warmup_stage) nags |= BIT(4);
00054 
00055     if (sendflags & BIT(6)) nags |= BIT(6);
00056 
00057     if (sendflags & BIT(7)) nags |= BIT(7);
00058 
00059     if (!(nags & 4))  // no vote called? send no string
00060         nags &= ~(BIT(6) | BIT(7));
00061 
00062     WriteByte(MSG_ENTITY, nags);
00063 
00064     if (nags & BIT(6))
00065     {
00066         WriteByte(MSG_ENTITY, vote_accept_count);
00067         WriteByte(MSG_ENTITY, vote_reject_count);
00068         WriteByte(MSG_ENTITY, vote_needed_overall);
00069         WriteChar(MSG_ENTITY, to.vote_selection);
00070     }
00071 
00072     if (nags & BIT(7)) WriteString(MSG_ENTITY, vote_called_display);
00073 
00074     if (nags & 1)
00075     {
00076         for (i = 1; i <= maxclients; i += 8)
00077         {
00078             for (f = 0, e = edict_num(i), b = 1; b < 256; b *= 2, e = nextent(e))
00079                 if (!IS_REAL_CLIENT(e) || e.ready) f |= b;
00080             WriteByte(MSG_ENTITY, f);
00081         }
00082     }
00083 
00084     return true;
00085 }

00205     float spectators_allowed = ((autocvar_sv_vote_nospectators != 2)

00223             case VOTE_SELECT_REJECT:
00224             { ++vote_reject_count;
00225               { if (IS_PLAYER(it)) ++vote_real_reject_count; } break;

00231             case VOTE_SELECT_ABSTAIN:
00232             { ++vote_abstain_count;
00233               { if (IS_PLAYER(it)) ++vote_real_abstain_count; } break;
00234             }
00235             default: break;
00236         }
00237     ));

00239     // Check to see if there are enough players on the server to allow master voting... otherwise, vote master could be used for evil.
00240     if ((vote_called == VOTE_MASTER) && autocvar_sv_vote_master_playerlimit > vote_player_count)
00241     {

00249     if (!spectators_allowed && (vote_real_player_count > 0))

00251         vote_accept_count = vote_real_accept_count;
00252         vote_reject_count = vote_real_reject_count;

00258     notvoters = (vote_player_count - vote_accept_count - vote_reject_count - vote_abstain_count);

00261     vote_factor_overall = bound(0.5, autocvar_sv_vote_majority_factor, 0.999);
00262     vote_needed_overall = floor((vote_player_count - vote_abstain_count) * vote_factor_overall) + 1;
00263 
00264     // if the vote times out, determine the amount of votes needed of the people who actually already voted
00265     vote_factor_of_voted = bound(0.5, autocvar_sv_vote_majority_factor_of_voted, 0.999);
00266     vote_needed_of_voted = floor((vote_accept_count + vote_reject_count) * vote_factor_of_voted) + 1;
00267 
00268     // are there any players at all on the server? it could be an admin vote
00269     if (vote_player_count == 0 && first_count)
00270     {
00271         VoteSpam(0, -1, "yes");  // no players at all, just accept it
00272         VoteAccept();
00273         return;
00274     }
00275 
00276     // since there ARE players, finally calculate the result of the vote
00277     if (vote_accept_count >= vote_needed_overall)
00278     {
00279         VoteSpam(notvoters, -1, "yes");  // there is enough acceptions to pass the vote
00280         VoteAccept();
00281         return;
00282     }
00283 
00284     if (vote_reject_count > vote_player_count - vote_abstain_count - vote_needed_overall)
00285     {
00286         VoteSpam(notvoters, -1, "no");  // there is enough rejections to deny the vote
00287         VoteReject();
00288         return;
00289     }
00290 
00291     // there is not enough votes in either direction, now lets just calculate what the voters have said
00292     if (time > vote_endtime)
00293     {
00294         final_needed_votes = vote_needed_overall;
00295 
00296         if (autocvar_sv_vote_majority_factor_of_voted)
00297         {
00298             if (vote_accept_count >= vote_needed_of_voted)
00299             {
00300                 VoteSpam(notvoters, min(vote_needed_overall, vote_needed_of_voted), "yes");
00301                 VoteAccept();
00302                 return;
00303             }
00304 
00305             if (vote_accept_count + vote_reject_count > 0)
00306             {
00307                 VoteSpam(notvoters, min(vote_needed_overall, vote_needed_of_voted), "no");
00308                 VoteReject();
00309                 return;
00310             }
00311 
00312             final_needed_votes = min(vote_needed_overall, vote_needed_of_voted);
00313         }
00314 
00315         // it didn't pass or fail, so not enough votes to even make a decision.
00316         VoteSpam(notvoters, final_needed_votes, "timeout");
00317         VoteTimeout();
00318     }
00319 }
