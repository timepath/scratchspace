721  for (i = 0; i < len; ++i)

724 
725  // if we're banned, don't even parse the command
726  if (Ban_MaybeEnforceBanOnce(this)) return;

728  float argc = tokenize_console(command);
729 
730  // Guide for working with argc arguments by example:
731  // argc: 1 - 2 - 3 - 4
732  // argv: 0 - 1 - 2 - 3
733  // cmd vote - master - login - password
734 
735  // for floodcheck
736  switch (strtolower(argv(0)))
737  {
738  // exempt commands which are not subject to floodcheck
739  case "begin": break; // handled by engine in host_cmd.c
740  case "download": break; // handled by engine in cl_parse.c
741  case "mv_getpicture": break; // handled by server in this file
742  case "pause": break; // handled by engine in host_cmd.c
743  case "prespawn": break; // handled by engine in host_cmd.c
744  case "sentcvar": break; // handled by server in this file
745  case "spawn": break; // handled by engine in host_cmd.c
746  case "c2s": Net_ClientCommand(this, command); return; // handled by net.qh
747 
748  default:
749  if (SV_ParseClientCommand_floodcheck(this)) break; // "true": continue, as we're not flooding yet
750  else return; // "false": not allowed to continue, halt // print("^1ERROR: ^7ANTISPAM CAUGHT: ", command, ".\n");
751  }
752 
753  /* NOTE: should this be disabled? It can be spammy perhaps, but hopefully it's okay for now */
754  if (argv(0) == "help")
