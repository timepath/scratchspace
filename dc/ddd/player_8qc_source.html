00990     // fix speedhacks :P

00992     // add the unused velocity as up component
00993     wishvel_z = 0;
00994

01000     float a_up = PHYS_JETPACK_ACCEL_UP(this);
01001     float a_add = PHYS_JETPACK_ANTIGRAVITY(this) * PHYS_GRAVITY(this);
01002

01004     wishvel_y *= a_side;
01005     wishvel_z *= a_up;

01011     // with wishvel having an absolute value of 1

01014     //   f(a_side, a_up, a_add, z) := a_side * (1 - z^2) + (a_add + a_up * z)^2;

01023         if (f > -1 && f < 1) // can it be attained?

01025             best = (a_diff + a_add * a_add) * (a_diff + a_up * a_up) / a_diff;
01026             //print("middle\n");

01028     }

01033         best = f;
01034         //print("top\n");
