00031 {
00032     vector vbest;
00033 
00034     if(v.x < 0) vbest.x = p.mins.x; else vbest.x = p.maxs.x;

00228 {
00229     float dist, distpersec, delta;
00230     vector v;
00231 
00232     dist = (eorg - porg) * pnorm;
00233     dist += min(emins.x * pnorm.x, emaxs.x * pnorm.x);
00234     dist += min(emins.y * pnorm.y, emaxs.y * pnorm.y);
00235     dist += min(emins.z * pnorm.z, emaxs.z * pnorm.z);
00236     if(dist < -1) // other side?

00243     delta = dist / distpersec;
00244     v = eorg - evel * delta - porg;
00245     v = v - pnorm * (pnorm * v);
