00739 void _MapInfo_Parse_Settemp(string pFilename, string acl, float type, string s, float recurse)
00740 {
00741     string t;
00742     float fh, o;
00743     t = car(s); s = cdr(s);
00744 
00745     // limited support of "" and comments
00746     //   remove trailing and leading " of t
00747     if(substring(t, 0, 1) == "\"")
00748     {
00749         if(substring(t, -1, 1) == "\"")
00750             t = substring(t, 1, -2);
00751     }
00752 
00753     //   remove leading " of s
00754     if(substring(s, 0, 1) == "\"")
00755     {
00756         s = substring(s, 1, -1);
00757     }
00758     //   remove trailing " of s, and all that follows (cvar description)
00759     o = strstrofs(s, "\"", 0);
00760     if(o >= 0)
00761         s = substring(s, 0, o);
00762 
00763     //   remove // comments
00764     o = strstrofs(s, "//", 0);
00765     if(o >= 0)
00766         s = substring(s, 0, o);
00767 
00768     //   remove trailing spaces
00769     while(substring(s, -1, 1) == " ")
00770         s = substring(s, 0, -2);
00771 
00772     if(t == "#include")
00773     {
00774         if(recurse > 0)
00775         {
00776             fh = fopen(s, FILE_READ);
00777             if(fh < 0)
00778                 LOG_MAPWARN("Map ", pFilename, " references not existing config file ", s, "\n");
00779             else
00780             {
00781                 for (;;)
00782                 {
00783                     if (!((s = fgets(fh))))
00784                         break;
00785 
00786                     // catch different sorts of comments
00787                     if(s == "")                    // empty lines
00788                         continue;
00789                     if(substring(s, 0, 1) == "#")  // UNIX style
00790                         continue;
00791                     if(substring(s, 0, 2) == "//") // C++ style
00792                         continue;
00793                     if(substring(s, 0, 1) == "_")  // q3map style
00794                         continue;
00795 
00796                     if(substring(s, 0, 4) == "set ")
00797                         s = substring(s, 4, -1);
00798                     if(substring(s, 0, 5) == "seta ")
00799                         s = substring(s, 5, -1);
00800 
00801                     _MapInfo_Parse_Settemp(pFilename, acl, type, s, recurse - 1);
00802                 }
00803                 fclose(fh);
00804             }
00805         }
00806         else
00807             LOG_MAPWARN("Map ", pFilename, " uses too many levels of inclusion\n");
00808     }
00809     else if(t == "")
00810         LOG_MAPWARN("Map ", pFilename, " contains a potentially harmful setting, ignored\n");
00811     else if (!cvar_value_issafe(t))
00812         LOG_MAPWARN("Map ", pFilename, " contains a potentially harmful setting, ignored\n");
00813     else if (!cvar_value_issafe(s))
00814         LOG_MAPWARN("Map ", pFilename, " contains a potentially harmful setting, ignored\n");
00815     else if(matchacl(MAPINFO_SETTEMP_ACL_SYSTEM, t) <= 0)
00816         LOG_MAPWARN("Map ", pFilename, " contains a potentially harmful setting, ignored\n");
00817     else if(matchacl(acl, t) <= 0)
00818         LOG_MAPWARN("Map ", pFilename, " contains a denied setting, ignored\n");
00819     else
00820     {
00821         if(type == 0) // server set
00822         {
00823             LOG_TRACE("Applying temporary setting ", t, " := ", s, "\n");
00824             if(cvar("g_campaign"))
00825                 cvar_set(t, s); // this is a wrapper and is always temporary anyway; no need to backup old values then
00826             else
00827                 cvar_settemp(t, s);
00828         }
00829         else
00830         {
00831             LOG_TRACE("Applying temporary client setting ", t, " := ", s, "\n");
00832             MapInfo_Map_clientstuff = strcat(
00833                     MapInfo_Map_clientstuff, "cl_cmd settemp \"", t, "\" \"", s, "\"\n"
00834                     );
00835         }
00836     }
00837 }

01179 float MapInfo_FindName(string s)
01180 {
01181     // if there is exactly one map of prefix s, return it
01182     // if not, return the null string
01183     // note that DP sorts glob results... so I can use a binary search
01184     float l, r, m, cmp;
01185     l = 0;
01186     r = MapInfo_count;
01187     // invariants: r is behind s, l-1 is equal or before
01188     while(l != r)
01189     {
01190         m = floor((l + r) / 2);
01191         MapInfo_FindName_match = _MapInfo_GlobItem(MapInfo_FilterList_Lookup(m));
01192         cmp = strcasecmp(MapInfo_FindName_match, s);
01193         if(cmp == 0)
01194             return m; // found and good
01195         if(cmp < 0)
01196             l = m + 1; // l-1 is before s
01197         else
01198             r = m; // behind s
01199     }
01200     MapInfo_FindName_match = _MapInfo_GlobItem(MapInfo_FilterList_Lookup(l));
01201     MapInfo_FindName_firstResult = l;
01202     // r == l, so: l is behind s, l-1 is before
01203     // SO: if there is any, l is the one with the right prefix
01204     //     and l+1 may be one too
01205     if(l == MapInfo_count)
01206     {
01207         MapInfo_FindName_match = string_null;
01208         MapInfo_FindName_firstResult = -1;
01209         return -1; // no MapInfo_FindName_match, behind last item
01210     }
01211     if(!startsWithNocase(MapInfo_FindName_match, s))
01212     {
01213         MapInfo_FindName_match = string_null;
01214         MapInfo_FindName_firstResult = -1;
01215         return -1; // wrong prefix
01216     }
01217     if(l == MapInfo_count - 1)
01218         return l; // last one, nothing can follow => unique
01219     if(startsWithNocase(_MapInfo_GlobItem(MapInfo_FilterList_Lookup(l + 1)), s))
01220     {
01221         MapInfo_FindName_match = string_null;
01222         return -1; // ambigous MapInfo_FindName_match
01223     }
01224     return l;
01225 }
