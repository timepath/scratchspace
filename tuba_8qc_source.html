00186 {
00187     float movestate = 5;
00188     if (pl.movement.x < 0)      movestate -= 3;
00189     else if (pl.movement.x > 0) movestate += 3;
00190     if (pl.movement.y < 0)      movestate -= 1;
00191     else if (pl.movement.y > 0) movestate += 1;
00192 
00193     int note = 0;
00194     switch (movestate)
00195     {

00198     //   B  c  d
00199     //   Gb G  G#
00200     // but then you only use forward and right key. So to make things more

00202     //   eb e  B
00203     // f=e# c  d
00204     //   Gb G  G#
00205         case 1: note = -6; break; // Gb
00206         case 2: note = -5; break; // G
00207         case 3: note = -4; break; // G#
00208         case 4: note = +5; break; // e#
00209         default:
00210         case 5: note =  0; break; // c
00211         case 6: note = +2; break; // d
00212         case 7: note = +3; break; // eb
00213         case 8: note = +4; break; // e
00214         case 9: note = -1; break; // B
00215     }
00216     if(pl.BUTTON_CROUCH)
00217         note -= 12;
00218     if(pl.BUTTON_JUMP)
00219         note += 12;
00220     if(hittype & HITTYPE_SECONDARY)
00221         note += 7;
00222 
00223     // we support two kinds of tubas, those tuned in Eb and those tuned in C
00224     // kind of tuba currently is player slot number, or team number if in
00225     // teamplay
00226     // that way, holes in the range of notes are "plugged"
00227     if(teamplay)
00228     {
00229         if(pl.team == NUM_TEAM_2 || pl.team == NUM_TEAM_4)
00230             note += 3;
00231     }
00232     else
00233     {
00234         if(pl.clientcolors & 1)
00235             note += 3;
00236     }
00237 
00238     // total range of notes:
00239     //                       0
00240     //                 ***  ** ****
00241     //                        ***  ** ****
00242     //     ***  ** ****
00243     //            ***  ** ****
00244     //     ***  ********************* ****

00246     //        ***  ********************* ****
