00989 {
00990     if (end == 0)

01000         // do the trace with respect to x

01144 float MoveToRandomLocationWithinBounds(entity e, vector boundmin, vector boundmax, float goodcontents, float badcontents, float badsurfaceflags, float attempts, float maxaboveground, float minviewdistance)
01145 {
01146     float m, i;
01147     vector start, org, delta, end, enddown, mstart;
01148     entity sp;
01149 
01150     m = e.dphitcontentsmask;
01151     e.dphitcontentsmask = goodcontents | badcontents;
01152 
01153     org = boundmin;
01154     delta = boundmax - boundmin;
01155 
01156     start = end = org;
01157 
01158     for (i = 0; i < attempts; ++i)
01159     {
01160         start.x = org.x + random() * delta.x;
01161         start.y = org.y + random() * delta.y;
01162         start.z = org.z + random() * delta.z;
01163 
01164         // rule 1: start inside world bounds, and outside
01165         // solid, and don't start from somewhere where you can
01166         // fall down to evil
01167         tracebox(start, e.mins, e.maxs, start - '0 0 1' * delta.z, MOVE_NORMAL, e);
01168         if (trace_fraction >= 1)
01169             continue;
01170         if (trace_startsolid)
01171             continue;
01172         if (trace_dphitcontents & badcontents)
01173             continue;
01174         if (trace_dphitq3surfaceflags & badsurfaceflags)
01175             continue;
01176 
01177         // rule 2: if we are too high, lower the point
01178         if (trace_fraction * delta.z > maxaboveground)
01179             start = trace_endpos + '0 0 1' * maxaboveground;
01180         enddown = trace_endpos;
01181 
01182         // rule 3: make sure we aren't outside the map. This only works
01183         // for somewhat well formed maps. A good rule of thumb is that
01184         // the map should have a convex outside hull.
01185         // these can be traceLINES as we already verified the starting box
01186         mstart = start + 0.5 * (e.mins + e.maxs);
01187         traceline(mstart, mstart + '1 0 0' * delta.x, MOVE_NORMAL, e);
01188         if (trace_fraction >= 1 || trace_dphittexturename == "common/caulk")
01189             continue;
01190         traceline(mstart, mstart - '1 0 0' * delta.x, MOVE_NORMAL, e);
01191         if (trace_fraction >= 1 || trace_dphittexturename == "common/caulk")
01192             continue;
