01009 {
01010     float i, imax;

01014     imax = 42;
01015 
01016     for(i = 0; i <= imax; ++i)

01018         float mapindex;

01026 float(float exponent) MaplistMethod_Shuffle = // more clever shuffling
01027 // the exponent sets a bias on the map selection:
01028 // the higher the exponent, the less likely "shortly repeated" same maps are
01029 {
01030     float i, j, imax, insertpos;
01031 
01032     LOG_TRACE("Trying MaplistMethod_Shuffle\n");
01033 
01034     imax = 42;
01035 
01036     for(i = 0; i <= imax; ++i)
01037     {
01038         string newlist;
01039 
01040         // now reinsert this at another position
01041         insertpos = pow(random(), 1 / exponent);       // ]0, 1]
01042         insertpos = insertpos * (Map_Count - 1);       // ]0, Map_Count - 1]
01043         insertpos = ceil(insertpos) + 1;               // {2, 3, 4, ..., Map_Count}
01044         LOG_TRACE("SHUFFLE: insert pos = ", ftos(insertpos), "\n");
01045 
01046         // insert the current map there
01047         newlist = "";
01048         for(j = 1; j < insertpos; ++j)                 // i == 1: no loop, will be inserted as first; however, i == 1 has been excluded above
01049             newlist = strcat(newlist, " ", argv(j));
01050         newlist = strcat(newlist, " ", argv(0));       // now insert the just selected map
01051         for(j = insertpos; j < Map_Count; ++j)         // i == Map_Count: no loop, has just been inserted as last
01052             newlist = strcat(newlist, " ", argv(j));
01053         newlist = substring(newlist, 1, strlen(newlist) - 1);
01054         cvar_set("g_maplist", newlist);
01055         Map_Count = tokenizebyseparator(autocvar_g_maplist, " ");
01056 
01057         // NOTE: the selected map has just been inserted at (insertpos-1)th position
01058         Map_Current = insertpos - 1; // this is not really valid, but this way the fallback has a chance of working
01059         if(Map_Check(Map_Current, 1))
01060             return Map_Current;
01061     }
01062     return -1;
01063 }
