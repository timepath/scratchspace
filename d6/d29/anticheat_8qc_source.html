00054 float movement_oddity(vector m0, vector m1)
00055 {
00056     float cosangle = normalize(m0) * normalize(m1);
00057     if(cosangle >= 0)
00058         return 0;
00059     return 0.5 - 0.5 * cos(cosangle * cosangle * 4 * M_PI);
00060         // returns 0 for: -1, -sqrt(0.5), 0 (angles that commonly happen with kbd)
