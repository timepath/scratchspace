00061     float update;

00063 
00064     if (delta == 0) return;
00065     if (autocvar_timelimit < 0) return;
00066 
00067     if (mi <= 10) mi = 10;  // at least ten sec in the future
00068     cur = time - game_starttime;
00069     if (cur > 0) mi += cur; // from current time!
00070 
00071     lim = autocvar_timelimit * 60;
00072 
00073     if (delta > 0)
00074     {
00075         if (lim == 0) return; // cannot increase any further
00076         else if (lim < ma) update = min(ma, lim + delta);

00078             return;

00080     else
00081     {
00082         if (lim == 0)      // infinite: try reducing to max, if we are allowed to
00083             update = max(mi, ma);
00084         else if (lim > mi) // above minimum: decrease
00085             update = max(mi, lim + delta);

00087             return;
00088     }

01365                 t_players = 0;
01366                 t_teams = 0;
01367                 FOREACH_CLIENT(IS_PLAYER(it) || it.caplayer, LAMBDA(
01368                     CheckAllowedTeams(it);
01369 
01370                     if (c1 >= 0) t_teams = max(1, t_teams);

01373                     if (c4 >= 0) t_teams = max(4, t_teams);

01377 
01378                 // build a list of the players in a random order
01379                 FOREACH_CLIENT(IS_PLAYER(it) || it.caplayer, LAMBDA(
01380                     for ( ; ; )
01381                     {
01382                         int idx = bound(1, floor(random() * maxclients) + 1, maxclients);
01383 
01384                         if (shuffleteams_players[idx])
01385                         {
01386                             continue;  // a player is already assigned to this slot
01387                         }
01388                         else
01389                         {
01390                             shuffleteams_players[idx] = etof(it);
01391                             break;
01392                         }
01393                     }
01394                 ));
01395 
01396                 // finally, from the list made earlier, re-join the players in different order.
01397                 for (int i = 1; i <= t_teams; ++i)
01398                 {
01399                     // find out how many players to assign to this team
01400                     x = (t_players / t_teams);
01401                     x = ((i == 1) ? ceil(x) : floor(x));
01402 
01403                     team_color = Team_NumberToTeam(i);
01404 
01405                     // sort through the random list of players made earlier
01406                     for (int z = 1; z <= maxclients; ++z)
01407                     {
01408                         if (!(shuffleteams_teams[i] >= x))
01409                         {
01410                             if (!(shuffleteams_players[z])) continue;  // not a player, move on to next random slot
01411 
01412                             if (VerifyClientNumber(shuffleteams_players[z])) setself(edict_num(shuffleteams_players[z]));
01413

01415 
01416                             shuffleteams_players[z] = 0;
01417                             shuffleteams_teams[i] = shuffleteams_teams[i] + 1;
01418                         }
01419                         else
01420                         {
01421                             break;  // move on to next team
01422                         }
01423                     }
