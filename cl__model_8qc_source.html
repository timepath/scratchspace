00134

00137         else

00139 
00140         l24_13 = this.csqcmodel_lerpfrac * llf + this.csqcmodel_lerpfrac2 * (1 - llf);

00142         this.lerpfrac  = l24 * l24_13;

00145 
00146         if(l24_13 == 0) // if frames 2/4 are not displayed, clear their frametime
00147         {
00148             this.frame2time = 0;
