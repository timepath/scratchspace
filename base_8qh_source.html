00053     {
00054         if (order & CBC_ORDER_FIRST) {
00055             if (order & CBC_ORDER_LAST)
00056                 if (this.cbc_order & CBC_ORDER_ANY)
00057                     return false;
00058             if (this.cbc_order & CBC_ORDER_FIRST)
00059                 return false;
00060         } else if (order & CBC_ORDER_LAST) {
00061             if (this.cbc_order & CBC_ORDER_LAST)
00062                 return false;
00063         }
00064         entity node = NEW(CallbackNode, cb, order);
00065         if (order & CBC_ORDER_FIRST) {
00066             node.cbc_next = this.cbc_next;
00067             this.cbc_next = node;
00068         } else if (order & CBC_ORDER_LAST) {
00069             CallbackNode prev = NULL, it = this.cbc_next;
00070             while (it) { prev = it, it = it.cbc_next; }
00071             if (prev) prev.cbc_next = node;
00072             else this.cbc_next = node;
00073         } else {
00074             // by default we execute last, but before a possible CBC_ORDER_LAST callback
00075             CallbackNode prev = NULL, it = this.cbc_next;
00076             while (it && !(it.cbc_order & CBC_ORDER_LAST)) { prev = it, it = it.cbc_next; }
00077             node.cbc_next = it;
00078             if (prev) prev.cbc_next = node;
00079             else this.cbc_next = node;
00080         }
00081         this.cbc_order |= (order | CBC_ORDER_ANY);
00082         return true;
