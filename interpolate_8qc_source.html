00129 void InterpolateOrigin_Do(entity this)
00130 {
00131     if (this.itime1 && this.itime2 && this.itime1 != this.itime2)
00132     {
00133         float f = bound(0, (time - this.itime1) / (this.itime2 - this.itime1), 1 + autocvar_cl_lerpexcess);
00134         float f_1 = 1 - f;
00135         if (this.iflags & IFLAG_ORIGIN) setorigin(this, f_1 * this.iorigin1 + f * this.iorigin2);
00136         if (this.iflags & IFLAG_ANGLES)
00137         {
00138             vector forward = f_1 * this.iforward1 + f * this.iforward2;
00139             vector up = f_1 * this.iup1 + f * this.iup2;
00140             this.angles = fixedvectoangles2(forward, up);
00141         }
00142         if (this.iflags & IFLAG_V_ANGLE)
00143         {
00144             vector forward = f_1 * this.ivforward1 + f * this.ivforward2;
00145             vector up = f_1 * this.ivup1 + f * this.ivup2;
00146             this.v_angle = fixedvectoangles2(forward, up);
00147         }
00148         else if (this.iflags & IFLAG_V_ANGLE_X)
00149         {
00150             this.v_angle_x = f_1 * this.ivforward1.x + f * this.ivforward2.x;
00151         }
00152         if (this.iflags & IFLAG_VELOCITY) this.velocity = f_1 * this.ivelocity1 + f * this.ivelocity2;
