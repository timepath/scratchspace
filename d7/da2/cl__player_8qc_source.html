217 vector CSQCPlayer_ApplySmoothing(entity this, vector v)
218 {
219  float smoothtime = bound(0, time - smooth_prevtime, 0.1);
220  smooth_prevtime = max(smooth_prevtime, drawtime); // drawtime is the previous frame's time at this point
221 
222  if(this.csqcmodel_teleported || !(this.pmove_flags & PMF_ONGROUND) || autocvar_cl_stairsmoothspeed <= 0)
223  stairsmoothz = v.z;
224  else
225  {
226  if(stairsmoothz < v.z)
227  v.z = stairsmoothz = bound(v.z - PHYS_STEPHEIGHT(this), stairsmoothz + smoothtime * autocvar_cl_stairsmoothspeed, v.z);
228  else if(stairsmoothz > v.z)
229  v.z = stairsmoothz = bound(v.z, stairsmoothz - smoothtime * autocvar_cl_stairsmoothspeed, v.z + PHYS_STEPHEIGHT(this));
230  }
231 
232  float viewheight = bound(0, (time - smooth_prevtime) / max(0.0001, autocvar_cl_smoothviewheight), 1);
233  viewheightavg = viewheightavg * (1 - viewheight) + this.view_ofs.z * viewheight;
234  v.z += viewheightavg;
235 
236  smooth_prevtime = time;
237 
238  return v;
239 }

282 vector CSQCPlayer_ApplyBobbing(entity this, vector v)
283 {
284  if(this.csqcmodel_isdead)
285  return v;
286 
287  // bounded XY speed, used by several effects below
288  float bob, cycle;
289 
290  // vertical view bobbing code
291  if(autocvar_cl_bob && autocvar_cl_bobcycle)
292  {
293  float bob_limit = autocvar_cl_bob_limit;
294 
295  if(autocvar_cl_bob_limit_heightcheck)
296  {
297  // use traces to determine what range the view can bob in, and scale down the bob as needed
298  vector bob_height_check_dest = v;
299  bob_height_check_dest.z += autocvar_cl_bob_limit * 1.1;
300  traceline(v, bob_height_check_dest, MOVE_NOMONSTERS, NULL);
301  float trace1fraction = trace_fraction;
302 
303  bob_height_check_dest = v;
304  bob_height_check_dest.z += autocvar_cl_bob_limit * -0.5;
305  traceline(v, bob_height_check_dest, MOVE_NOMONSTERS, NULL);
306  float trace2fraction = trace_fraction;
307 
308  bob_limit *= min(trace1fraction, trace2fraction);
309  }
310 
311  // LordHavoc: figured out bobup: the time at which the sin is at 180
312  // degrees (which allows lengthening or squishing the peak or valley)
313  cycle = time / autocvar_cl_bobcycle;
314  cycle -= rint(cycle);
315  if(cycle < autocvar_cl_bobup)
316  cycle = sin(M_PI * cycle / autocvar_cl_bobup);
317  else
318  cycle = sin(M_PI + M_PI * (cycle - autocvar_cl_bobup) / (1.0 - autocvar_cl_bobup));
319  // bob is proportional to velocity in the xy plane
320  // (don't count Z, or jumping messes it up)
321  float xyspeed = bound(0, sqrt(this.velocity.x * this.velocity.x + this.velocity.y * this.velocity.y), autocvar_cl_bob_velocity_limit);
322  bob = xyspeed * autocvar_cl_bob;
323  bob = bound(0, bob, bob_limit);
324  bob = bob * 0.3 + bob * 0.7 * cycle;
325  v.z += bob;
326  }
327 
328  // horizontal view bobbing code
329  if(autocvar_cl_bob2 && autocvar_cl_bob2cycle)
330  {
331  cycle = time / autocvar_cl_bob2cycle;
332  cycle -= rint(cycle);
333  if(cycle < 0.5)
334  cycle = cos(M_PI * cycle / 0.5); // cos looks better here with the other view bobbing using sin
335  else
336  cycle = cos(M_PI + M_PI * (cycle - 0.5) / 0.5);
337  bob = autocvar_cl_bob2 * cycle;
338 
339  // this value slowly decreases from 1 to 0 when we stop touching the ground.
340  // The cycle is later multiplied with it so the view smooths back to normal
341  if(IS_ONGROUND(this) && !(input_buttons & BIT(1))) // also block the effect while the jump button is pressed, to avoid twitches when bunny-hopping
342  bob2_smooth = 1;
343  else
344  {
345  if(bob2_smooth > 0)
346  bob2_smooth -= bound(0, autocvar_cl_bob2smooth, 1);
347  else
348  bob2_smooth = 0;
349  }
350 
351  // calculate the front and side of the player between the X and Y axes
352  makevectors(view_angles);
353  // now get the speed based on those angles. The bounds should match the same value as xyspeed's
354  float side = bound(-autocvar_cl_bob_velocity_limit, (this.velocity * v_right) * bob2_smooth, autocvar_cl_bob_velocity_limit);
355  float front = bound(-autocvar_cl_bob_velocity_limit, (this.velocity * v_forward) * bob2_smooth, autocvar_cl_bob_velocity_limit);
356  v_forward = v_forward * bob;
357  v_right = v_right * bob;
358  // we use side with forward and front with right, so the bobbing goes
359  // to the side when we walk forward and to the front when we strafe
360  vector bob2vel;
361  bob2vel.x = side * v_forward.x + front * v_right.x + 0 * v_up.x;
362  bob2vel.y = side * v_forward.y + front * v_right.y + 0 * v_up.y;
363  bob2vel.z = side * v_forward.z + front * v_right.z + 0 * v_up.z;
364  v.x += bob2vel.x;
365  v.y += bob2vel.y;
366  }
367 
368  // fall bobbing code
369  // causes the view to swing down and back up when touching the ground
370  if(autocvar_cl_bobfall && autocvar_cl_bobfallcycle)
371  {
372  if(!IS_ONGROUND(this))
373  {
374  bobfall_speed = bound(-400, this.velocity.z, 0) * bound(0, autocvar_cl_bobfall, 0.1);
375  if(this.velocity.z < -autocvar_cl_bobfallminspeed)
376  bobfall_swing = 1;
377  else
378  bobfall_swing = 0; // really?
379  }
380  else
381  {
382  bobfall_swing = max(0, bobfall_swing - autocvar_cl_bobfallcycle * frametime);
383  float bobfall = sin(M_PI * bobfall_swing) * bobfall_speed;
384  v.z += bobfall;
385  }
386  }
387 
388  return v;
389 }

393 float CSQCPlayer_CalcRoll(entity this)
394 {
395  makevectors(view_angles);
396  float side = (this.velocity * v_right);
397  float sign = (side < 0) ? -1 : 1;
398  side = fabs(side);
399 
400  if(side < autocvar_cl_rollspeed)
401  side = side * autocvar_cl_rollangle / autocvar_cl_rollspeed;
402  else
403  side = autocvar_cl_rollangle;
404 
405  return side * sign;
406 }

412 vector CSQCPlayer_ApplyChase(entity this, vector v)
413 {
414  vector forward;
415  vector chase_dest;
416 
417  if(autocvar_chase_overhead)
418  {
419  view_angles.x = 0;
420  makevectors(view_angles);
421  forward = v_forward;
422  vector up = v_up;
423  // trace a little further so it hits a surface more consistently (to avoid 'snapping' on the edge of the range)
424  chase_dest.x = v.x - forward.x * autocvar_chase_back + up.x * autocvar_chase_up;
425  chase_dest.y = v.y - forward.y * autocvar_chase_back + up.y * autocvar_chase_up;
426  chase_dest.z = v.z - forward.z * autocvar_chase_back + up.z * autocvar_chase_up;
427 
428  // trace from first person view location to our chosen third person view location
429  traceline(v, chase_dest, MOVE_NOMONSTERS, NULL);
430 
431  vector bestvieworg = trace_endpos;
432  vector offset = '0 0 0';
433  for(offset.x = -16; offset.x <= 16; offset.x += 8)
434  {
435  for(offset.y = -16; offset.y <= 16; offset.y += 8)
436  {
437  makevectors(view_angles);
438  up = v_up;
439  chase_dest.x = v.x - forward.x * autocvar_chase_back + up.x * autocvar_chase_up + offset.x;
440  chase_dest.y = v.y - forward.y * autocvar_chase_back + up.y * autocvar_chase_up + offset.y;
441  chase_dest.z = v.z - forward.z * autocvar_chase_back + up.z * autocvar_chase_up + offset.z;
442  traceline(v, chase_dest, MOVE_NOMONSTERS, NULL);
443  if(bestvieworg.z > trace_endpos.z)
444  bestvieworg.z = trace_endpos.z;
445  }
446  }
447  bestvieworg.z -= 8;
448  v = bestvieworg;
449 
450  view_angles.x = autocvar_chase_pitchangle;
451  //setproperty(VF_CL_VIEWANGLES, view_angles); // update view angles as well so we can aim
452  }
453  else
454  {
455  makevectors(view_angles);
456  forward = v_forward;
457  // trace a little further so it hits a surface more consistently (to avoid 'snapping' on the edge of the range)
458  float cdist = -autocvar_chase_back - 8;
459  chase_dest.x = v.x + forward.x * cdist;
460  chase_dest.y = v.y + forward.y * cdist;
461  chase_dest.z = v.z + forward.z * cdist + autocvar_chase_up;
462  traceline(v, chase_dest, MOVE_NOMONSTERS, NULL);
463  v.x = 1 * trace_endpos.x + 8 * forward.x + 4 * trace_plane_normal.x;
464  v.y = 1 * trace_endpos.y + 8 * forward.y + 4 * trace_plane_normal.y;
465  v.z = 1 * trace_endpos.z + 8 * forward.z + 4 * trace_plane_normal.z;
466  }
467 
468 #if 0
469  tracebox(v, '-4 -4 -4', '4 4 4', v - v_forward * autocvar_chase_back, MOVE_NORMAL, this);
470  v = trace_endpos;
471  tracebox(v, '-4 -4 -4', '4 4 4', v + v_up * autocvar_chase_up, MOVE_NORMAL, this);
472  v = trace_endpos;
473 #endif
474  return v;
475 }

510 void CSQCPlayer_SetCamera()
511 {
512  vector v0 = ((intermission && !autocvar_cl_movement_intermissionrunning) ? '0 0 0' : pmove_vel); // TRICK: pmove_vel is set by the engine when we get here. No need to network velocity
513  float vh = PHYS_VIEWHEIGHT(NULL);

515  vector pl_viewofs_crouch = PHYS_PL_CROUCH_VIEWOFS(NULL);
516  entity e = csqcplayer;
517  if (e)
518  {
519  if (servercommandframe == 0 || clientcommandframe == 0)
520  {
521  InterpolateOrigin_Do(e);
522  e.view_ofs = '0 0 1' * vh;
523 
524  // get crouch state from the server
525  if (vh == pl_viewofs.z) e.flags &= ~FL_DUCKED;
526  else if (vh == pl_viewofs_crouch.z) e.flags |= FL_DUCKED;
527 
528  // get onground state from the server
529  e.flags = BITSET(e.flags, FL_ONGROUND, pmove_onground);
530 
531  CSQCPlayer_SetMinsMaxs(e);
532 
533  // override it back just in case
534  e.view_ofs = '0 0 1' * vh;
535 
536  // set velocity
537  e.velocity = v0;
538  }
539  else
540  {
541  int flg = e.iflags; e.iflags &= ~(IFLAG_ORIGIN | IFLAG_ANGLES);
542  InterpolateOrigin_Do(e);
543  e.iflags = flg;
544 
545  if (csqcplayer_status == CSQCPLAYERSTATUS_FROMSERVER)
546  {
547  vector o = e.origin;
548  csqcplayer_status = CSQCPLAYERSTATUS_PREDICTED;
549  CSQCPlayer_PredictTo(e, servercommandframe + 1, false);
550  CSQCPlayer_SetPredictionError(e.origin - o, e.velocity - v0, pmove_onground - IS_ONGROUND(e));
551  e.origin = o;
552  e.velocity = v0;
553 
554  // get crouch state from the server
555  if (vh == pl_viewofs.z) e.flags &= ~FL_DUCKED;
556  else if(vh == pl_viewofs_crouch.z) e.flags |= FL_DUCKED;
557 
558  // get onground state from the server
559  e.flags = BITSET(e.flags, FL_ONGROUND, pmove_onground);
560 
561  CSQCPlayer_SavePrediction(e);
562  }
563  CSQCPlayer_PredictTo(e, clientcommandframe + 1, true);
564 
565 #ifdef CSQCMODEL_SERVERSIDE_CROUCH
566  // get crouch state from the server (LAG)
567  if (vh == pl_viewofs.z) e.flags &= ~FL_DUCKED;
