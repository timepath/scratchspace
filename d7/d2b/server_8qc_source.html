00616     for(i_s = 0; ; ++i_s)
00617     {
00618         tex = getsurfacetexture(self, i_s);

00624         for(i_t = 0; i_t < n_t; ++i_t)
00625         {

00627             a = getsurfacepoint(self, i_s, tri.x);
00628             b = getsurfacepoint(self, i_s, tri.y);

00630             n = cross(c - a, b - a);
00631             area = area + vlen(n);
00632             norm = norm + n;
00633             point = point + vlen(n) * (a + b + c);
00634         }
00635     }
00636     if(area > 0)
00637     {
00638         norm = norm * (1 / area);
00639         point = point * (1 / (3 * area));
00640         if(vdist(norm, <, 0.99))
00641         {
00642             LOG_INFO("trigger_warpzone near ", vtos(self.aiment.origin), " is nonplanar. BEWARE.\n");
00643             area = 0; // no autofixing in this case
00644         }
00645         norm = normalize(norm);
00646     }
00647 
00648     ang = '0 0 0';
00649     if(self.aiment)
00650     {
00651         org = self.aiment.origin;
00652         ang = self.aiment.angles;
00653         if(area > 0)
00654         {
00655             org = org - ((org - point) * norm) * norm; // project to plane
00656             makevectors(ang);
00657             if(norm * v_forward < 0)
00658             {
00659                 LOG_INFO("Position target of trigger_warpzone near ", vtos(self.aiment.origin), " points into trigger_warpzone. BEWARE.\n");
00660                 norm = -1 * norm;
00661             }
00662             ang = vectoangles2(norm, v_up); // keep rotation, but turn exactly against plane
00663             ang.x = -ang.x;
00664             if(norm * v_forward < 0.99)
00665                 LOG_INFO("trigger_warpzone near ", vtos(self.aiment.origin), " has been turned to match plane orientation (", vtos(self.aiment.angles), " -> ", vtos(ang), "\n");
00666             if(vdist(org - self.aiment.origin, >, 0.5))

00670     else if(area > 0)
00671     {
00672         org = point;
00673         ang = vectoangles(norm);

00675     }
00676     else
00677         error("cannot infer origin/angles for this warpzone, please use a killtarget or a trigger_warpzone_position");
