00890 // return value:
00891 //    1 = WP created
00892 //    0 = no action needed
00893 //   -1 = temp fail, try from world too
00894 //   -2 = permanent fail, do not retry
00895 float botframe_autowaypoints_fix_from(entity p, float walkfromwp, entity wp, .entity fld)
00896 {
00897     // make it possible to go from p to wp, if we can
00898     // if wp is world, nearest is chosen
00899 
00900     entity w;
00901     vector porg;
00902     float t, tmin, tmax;
00903     vector o;
00904     vector save;
00905 
00906     if(!botframe_autowaypoints_fixdown(p.origin))
00907         return -2;
00908     porg = trace_endpos;
00909 
00910     if(wp)
00911     {
00912         // if any WP w fulfills wp -> w -> porg and w is closer than wp, then switch from wp to w
00913 
00914         // if wp -> porg, then OK
00915         float maxdist;
00916         if(navigation_waypoint_will_link(wp.origin, porg, p, walkfromwp, 1050))
00917         {
00918             // we may find a better one
00919             maxdist = vlen(wp.origin - porg);
00920         }
00921         else
00922         {
00923             // accept any "good"
00924             maxdist = 2100;
00925         }
00926 
00927         float bestdist;
00928         bestdist = maxdist;
00929         w = find(world, classname, "waypoint");
00930         while (w)
00931         {
00932             if(w != wp && !(w.wpflags & WAYPOINTFLAG_NORELINK))
00933             {
00934                 float d;
00935                 d = vlen(wp.origin - w.origin) + vlen(w.origin - porg);
00936                 if(d < bestdist)
00937                     if(navigation_waypoint_will_link(wp.origin, w.origin, p, walkfromwp, 1050))
00938                         if(navigation_waypoint_will_link(w.origin, porg, p, walkfromwp, 1050))
00939                         {
00940                             bestdist = d;
00941                             p.(fld) = w;
00942                         }
00943             }
00944             w = find(w, classname, "waypoint");
00945         }
00946         if(bestdist < maxdist)
00947         {
00948             LOG_INFO("update chain to new nearest WP ", etos(p.(fld)), "\n");
00949             return 0;
00950         }
00951 
00952         if(bestdist < 2100)
00953         {
00954             // we know maxdist < 2100
00955             // so wp -> porg is still valid
00956             // all is good
00957             p.(fld) = wp;
00958             return 0;
00959         }
00960 
00961         // otherwise, no existing WP can fix our issues
00962     }
00963     else
00964     {
00965         save = p.origin;
00966         setorigin(p, porg);
00967         w = navigation_findnearestwaypoint(p, walkfromwp);
00968         setorigin(p, save);
00969         if(w)
00970         {
00971             p.(fld) = w;
00972             return 0;
00973         }
00974     }
00975 
00976     tmin = 0;
00977     tmax = 1;
00978     for (;;)
00979     {
00980         if(tmax - tmin < 0.001)
00981         {
00982             // did not get a good candidate
00983             return -1;
00984         }
00985 
00986         t = (tmin + tmax) * 0.5;
00987         o = antilag_takebackorigin(p, time - t);
00988         if(!botframe_autowaypoints_fixdown(o))
00989             return -2;
00990         o = trace_endpos;
00991 
00992         if(wp)
00993         {
00994             if(!navigation_waypoint_will_link(wp.origin, o, p, walkfromwp, 1050))
00995             {
00996                 // we cannot walk from wp.origin to o
00997                 // get closer to tmax
00998                 tmin = t;
00999                 continue;
01000             }
01001         }
01002         else
01003         {
01004             save = p.origin;
01005             setorigin(p, o);
01006             w = navigation_findnearestwaypoint(p, walkfromwp);
01007             setorigin(p, save);
01008             if(!w)
01009             {
01010                 // we cannot walk from any WP to o
01011                 // get closer to tmax
01012                 tmin = t;
01013                 continue;
01014             }
01015         }
01016 
01017         // if we get here, o is valid regarding waypoints
01018         // check if o is connected right to the player
01019         // we break if it succeeds, as that means o is a good waypoint location
01020         if(navigation_waypoint_will_link(o, porg, p, walkfromwp, 1050))
01021             break;
01022 
01023         // o is no good, we need to get closer to the player
01024         tmax = t;
01025     }
01026 
01027     LOG_INFO("spawning a waypoint for connecting to ", etos(wp), "\n");
01028     botframe_autowaypoints_createwp(o, p, fld, 0);
01029     return 1;
01030 }
