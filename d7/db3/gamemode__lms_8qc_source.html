00074 // LMS winning condition: game terminates if and only if there's at most one
00075 // one player who's living lives. Top two scores being equal cancels the time
00076 // limit.
00077 int WinningCondition_LMS()
00078 {
00079     entity head, head2;
00080     bool have_player = false;
00081     bool have_players = false;
00082 
00083     int l = LMS_NewPlayerLives();
00084 
00085     head = find(world, classname, STR_PLAYER);
00086     if(head)
00087         have_player = true;
00088     head2 = find(head, classname, STR_PLAYER);
00089     if(head2)
00090         have_players = true;
00091 
00092     if(have_player)
00093     {
00094         // we have at least one player
00095         if(have_players)
00096         {
00097             // two or more active players - continue with the game
00098         }
00099         else
00100         {
00101             // exactly one player?
00102 
00103             ClearWinners();
00104             SetWinners(winning, 0); // NOTE: exactly one player is still "player", so this works out
00105 
00106             if(l)
00107             {
00108                 // game still running (that is, nobody got removed from the game by a frag yet)? then continue
00109                 return WINNING_NO;
00110             }
00111             else
00112             {
00113                 // a winner!
00114                 // and assign him his first place
00115                 PlayerScore_Add(head, SP_LMS_RANK, 1);
00116                 return WINNING_YES;
00117             }
00118         }
00119     }
00120     else
00121     {
00122         // nobody is playing at all...
00123         if(l)
00124         {
00125             // wait for players...
00126         }
00127         else
00128         {
00129             // SNAFU (maybe a draw game?)
00130             ClearWinners();
00131             LOG_TRACE("No players, ending game.\n");
00132             return WINNING_YES;
00133         }
00134     }
00135 
00136     // When we get here, we have at least two players who are actually LIVING,
00137     // now check if the top two players have equal score.
00138     WinningConditionHelper();
00139 
00140     ClearWinners();
00141     if(WinningConditionHelper_winner)
00142         WinningConditionHelper_winner.winning = true;
00143     if(WinningConditionHelper_topscore == WinningConditionHelper_secondscore)
00144         return WINNING_NEVER;
00145 
00146     // Top two have different scores? Way to go for our beloved TIMELIMIT!
00147     return WINNING_NO;
00148 }
