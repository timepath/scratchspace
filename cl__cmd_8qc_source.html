00536 
00537     // Guide for working with argc arguments by example:
00538     // argc:   1    - 2      - 3     - 4

00540     // cmd     vote - master - login - password
00541     string s = strtolower(argv(0));

00553             LOG_INFO("For help about a specific command, type cl_cmd help COMMAND\n");
00554 
00555             return;
00556         }
00557         else if (GenericCommand_macro_usage(argc))  // Instead of trying to call a command, we're going to see detailed information about it
00558         {
00559             return;
00560         }
00561         else if (LocalCommand_macro_usage(argc))  // now try for normal commands too
00562         {
00563             return;
00564         }
00565     }
00566     // continue as usual and scan for normal commands
00567     if (GenericCommand(command)                                    // handled by common/command/generic.qc
00568         || LocalCommand_macro_command(argc, command)               // handled by one of the above LocalCommand_* functions
00569         || MUTATOR_CALLHOOK(CSQC_ConsoleCommand, s, argc, command) // handled by a mutator
00570        ) return;
00571 
00572     // nothing above caught the command, must be invalid
00573     LOG_INFO(((command != "") ? strcat("Unknown client command \"", command, "\"") : "No command provided"), ". For a list of supported commands, try cl_cmd help.\n");
