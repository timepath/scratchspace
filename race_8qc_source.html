01147 float race_GetFractionalLapCount(entity e)
01148 {
01149     // interesting metrics (idea by KrimZon) to maybe sort players in the
01150     // scoreboard, immediately updates when overtaking
01151     //
01152     // requires the track to be built so you never get farther away from the
01153     // next checkpoint, though, and current Xonotic race maps are not built that
01154     // way
01155     //
01156     // also, this code is slow and would need optimization (i.e. "next CP"
01157     // links on CP entities)
01158 
01159     float l;
01160     l = PlayerScore_Add(e, SP_RACE_LAPS, 0);
01161     if(e.race_completed)
01162         return l; // not fractional
01163 
01164     vector o0, o1;
01165     float bestfraction, fraction;
01166     entity lastcp, cp0, cp1;
01167     float nextcpindex, lastcpindex;
01168 
01169     nextcpindex = max(e.race_checkpoint, 0);
01170     lastcpindex = e.race_respawn_checkpoint;
01171     lastcp = e.race_respawn_spotref;
01172 
01173     if(nextcpindex == lastcpindex)
01174         return l; // finish
01175 
01176     bestfraction = 1;
01177     for(cp0 = world; (cp0 = find(cp0, classname, "trigger_race_checkpoint")); )
01178     {
01179         if(cp0.race_checkpoint != lastcpindex)
01180             continue;
01181         if(lastcp)
01182             if(cp0 != lastcp)
01183                 continue;
01184         o0 = (cp0.absmin + cp0.absmax) * 0.5;
01185         for(cp1 = world; (cp1 = find(cp1, classname, "trigger_race_checkpoint")); )
01186         {
01187             if(cp1.race_checkpoint != nextcpindex)
01188                 continue;
01189             o1 = (cp1.absmin + cp1.absmax) * 0.5;
01190             if(o0 == o1)
01191                 continue;
01192             fraction = bound(0.0001, vlen(e.origin - o1) / vlen(o0 - o1), 1);
01193             if(fraction < bestfraction)
01194                 bestfraction = fraction;
01195         }
01196     }
01197 
01198     // we are at CP "nextcpindex - bestfraction"
01199     // race_timed_checkpoint == 4: then nextcp==4 means 0.9999x, nextcp==0 means 0.0000x
01200     // race_timed_checkpoint == 0: then nextcp==0 means 0.9999x
01201     float c, nc;
01202     nc = race_highest_checkpoint + 1;
01203     c = ((nextcpindex - race_timed_checkpoint + nc + nc - 1) % nc) + 1 - bestfraction;
01204 
01205     return l + c / nc;
01206 }
