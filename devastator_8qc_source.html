00222 vector W_Devastator_SteerTo(vector thisdir, vector goaldir, float maxturn_cos)
00223 {
00224     if(thisdir * goaldir > maxturn_cos)
00225         return goaldir;
00226     if(thisdir * goaldir < -0.9998) // less than 1 degree and opposite
00227         return thisdir; // refuse to guide (better than letting a numerical error happen)
00228     float f, m2;
00229     vector v;
00230     // solve:
00231     //   g = normalize(thisdir + goaldir * X)
00232     //   thisdir * g = maxturn
00233     //
00234     //   gg = thisdir + goaldir * X
00235     //   (thisdir * gg)^2 = maxturn^2 * (gg * gg)
00236     //
00237     //   (1 + (thisdir * goaldir) * X)^2 = maxturn^2 * (1 + X*X + 2 * X * thisdir * goaldir)
00238     f = thisdir * goaldir;
00239     //   (1 + f * X)^2 = maxturn^2 * (1 + X*X + 2 * X * f)
00240     //   0 = (m^2 - f^2) * x^2 + (2 * f * (m^2 - 1)) * x + (m^2 - 1)
00241     m2 = maxturn_cos * maxturn_cos;
00242     v = solve_quadratic(m2 - f * f, 2 * f * (m2 - 1), m2 - 1);
00243     return normalize(thisdir + goaldir * v.y); // the larger solution!
00244 }

00246 //   f == -1
00247 //   v = solve_qadratic(m2 - 1, -2 * (m2 - 1), m2 - 1)

00249 //   x^2 - 2 * x + 1 = 0
00250 //   (x - 1)^2 = 0
