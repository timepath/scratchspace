00092         if(ratio > item_aspect)

00094 
00095         if(ratio <= best_ratio)
00096             break; // ratio starts decreasing by now, skip next configurations
00097

00099         best_rows = rows;
00100         best_ratio = ratio;
00101 
00102         if(columns == 1)
00103             break;
00104 
00105         --columns;
00106         rows = ceil(item_count/columns);
00107     }
00108 
00109     if(vertical)
00110         return eX * best_rows + eY * best_columns;
00111     else
00112         return eX * best_columns + eY * best_rows;
