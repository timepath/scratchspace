00962 vector solve_shotdirection(vector myorg, vector myvel, vector eorg, vector evel, float spd, float newton_style)
00963 {
00964     vector ret;
00965 
00966     // make origin and speed relative
00967     eorg -= myorg;
00968     if(newton_style)
00969         evel -= myvel;
00970 
00971     // now solve for ret, ret normalized:
00972     //   eorg + t * evel == t * ret * spd
00973     // or, rather, solve for t:
00974     //   |eorg + t * evel| == t * spd
00975     //   eorg^2 + t^2 * evel^2 + 2 * t * (eorg * evel) == t^2 * spd^2
00976     //   t^2 * (evel^2 - spd^2) + t * (2 * (eorg * evel)) + eorg^2 == 0
00977     vector solution = solve_quadratic(evel * evel - spd * spd, 2 * (eorg * evel), eorg * eorg);
00978     // p = 2 * (eorg * evel) / (evel * evel - spd * spd)
00979     // q = (eorg * eorg) / (evel * evel - spd * spd)
00980     if(!solution.z) // no real solution
00981     {
00982         // happens if D < 0
00983         // (eorg * evel)^2 < (evel^2 - spd^2) * eorg^2
00984         // (eorg * evel)^2 / eorg^2 < evel^2 - spd^2
00985         // spd^2 < ((evel^2 * eorg^2) - (eorg * evel)^2) / eorg^2
00986         // spd^2 < evel^2 * (1 - cos^2 angle(evel, eorg))
00987         // spd^2 < evel^2 * sin^2 angle(evel, eorg)
00988         // spd < |evel| * sin angle(evel, eorg)
00989         return '0 0 0';
00990     }
00991     else if(solution.x > 0)
00992     {
00993         // both solutions > 0: take the smaller one
00994         // happens if p < 0 and q > 0
00995         ret = normalize(eorg + solution.x * evel);
00996     }
00997     else if(solution.y > 0)
00998     {
00999         // one solution > 0: take the larger one
01000         // happens if q < 0 or q == 0 and p < 0
01001         ret = normalize(eorg + solution.y * evel);
01002     }
01003     else
01004     {
01005         // no solution > 0: reject
01006         // happens if p > 0 and q >= 0
01007         // 2 * (eorg * evel) / (evel * evel - spd * spd) > 0
01008         // (eorg * eorg) / (evel * evel - spd * spd) >= 0
01009         //
01010         // |evel| >= spd
01011         // eorg * evel > 0
01012         //
01013         // "Enemy is moving away from me at more than spd"
01014         return '0 0 0';
01015     }
01016 
01017     // NOTE: we always got a solution if spd > |evel|
01018 
01019     if(newton_style == 2)
01020         ret = normalize(ret * spd + myvel);
01021 
01022     return ret;
01023 }

01413     // queue_end is the last entity added

01427         entity t;
