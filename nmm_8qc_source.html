00387 // whether the given tile is a valid selection

00389 {
00390     if ( ( tile.owner.minigame_flags & NMM_TURN_TEAM ) != minigame_self.team )
00391         return false; // not our turn
00392     if ( tile.owner.minigame_flags & NMM_TURN_PLACE )
00393         return !tile.nmm_tile_piece; // need to put a piece on an empty spot
00394     if ( tile.owner.minigame_flags & NMM_TURN_MOVE )
00395     {
00396         if ( tile.nmm_tile_piece && tile.nmm_tile_piece.team == minigame_self.team &&

00400             return !tile.nmm_tile_piece && nmm_tile_adjacent(nmm_fromtile,tile);
00401         return false;
00402     }

00404     {
00405         if ( nmm_fromtile )
00406             return !tile.nmm_tile_piece;

00408             return tile.nmm_tile_piece && tile.nmm_tile_piece.team == minigame_self.team;
00409     }
00410     if ( tile.owner.minigame_flags & NMM_TURN_TAKE )
00411         return tile.nmm_tile_piece && tile.nmm_tile_piece.team != minigame_self.team &&
