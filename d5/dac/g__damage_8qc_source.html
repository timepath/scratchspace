01110         {
01111             // Constraints:
01112

01114             mindamage = mindps * mintime;
01115 
01116             // damage we want to get
01117             maxdamage = mindamage + d;
01118 
01119             // but we can't exceed maxtime * maxdps!
01120             totaldamage = min(maxdamage, maxtime * maxdps);
01121 
01122             // LEMMA:
01123             // Look at:
01124             // totaldamage = min(mindamage + d, maxtime * maxdps)
01125             // We see:
01126             // totaldamage <= maxtime * maxdps
01127             // ==> totaldamage / maxdps <= maxtime.
01128             // We also see:
01129             // totaldamage / mindps = min(mindamage / mindps + d, maxtime * maxdps / mindps)
01130             //                     >= min(mintime, maxtime)
01131             // ==> totaldamage / maxdps >= mintime.
01132 
01133             /*
01134             // how long do we damage then?
01135             // at least as long as before

01138             */
01139 
01140             // alternate:
01141             // at most as long as maximum allowed
01142             // but, never below mindps
01143             totaltime = min(maxtime, totaldamage / mindps); // always >= mintime due to lemma
01144 
01145             // assuming t > mintime, dps > mindps:
01146             // we get d = t * dps = maxtime * maxdps
01147             // totaldamage = min(maxdamage, maxtime * maxdps) = min(... + d, maxtime * maxdps) = maxtime * maxdps
01148             // totaldamage / maxdps = maxtime

01152             // b) totaltime = min(maxtime, totaldamage / maxdps) = maxtime

01156             // a) totaltime = max(mintime, ...) >= mintime, also totaltime <= maxtime by the lemma, therefore totaltime = mintime = maxtime
01157             // b) totaltime = min(maxtime, ...) <= maxtime, also totaltime >= mintime by the lemma, therefore totaltime = mintime = maxtime

01160             // we get mindps = maxdps.
