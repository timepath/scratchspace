657 void Draw_Shockwave(entity this)
658 {
659  // fading/removal control
660  float a = bound(0, (SW_MAXALPHA - ((time - this.sw_time) / SW_FADETIME)), SW_MAXALPHA);
661  if(a < ALPHA_MIN_VISIBLE) { delete(this); }
662 
663  // WEAPONTODO: save this only once when creating the entity
664  vector sw_color = entcs_GetColor(this.sv_entnum - 1); // Team_ColorRGB(entcs_GetTeam(this.sv_entnum));
665 
666  // WEAPONTODO: trace to find what we actually hit
667  vector endpos = (this.sw_shotorg + (this.sw_shotdir * this.sw_distance));
668 
669  vectorvectors(this.sw_shotdir);
670  vector right = v_right; // save this for when we do makevectors later
671  vector up = v_up; // save this for when we do makevectors later
672 
673  // WEAPONTODO: combine and simplify these calculations
674  vector min_end = ((this.sw_shotorg + (this.sw_shotdir * SW_DISTTOMIN)) + (up * this.sw_spread_min));
675  vector max_end = (endpos + (up * this.sw_spread_max));
676  float spread_to_min = vlen(normalize(min_end - this.sw_shotorg) - this.sw_shotdir);
677  float spread_to_max = vlen(normalize(max_end - min_end) - this.sw_shotdir);
678 
679  vector first_min_end = '0 0 0', prev_min_end = '0 0 0', new_min_end = '0 0 0';
680  vector first_max_end = '0 0 0', prev_max_end = '0 0 0', new_max_end = '0 0 0';
681  float new_max_dist, new_min_dist;
682 
683  vector deviation, angle = '0 0 0';
684  float counter, divisions = 20;
685  for(counter = 0; counter < divisions; ++counter)
686  {
687  // perfect circle effect lines
688  makevectors('0 360 0' * (0.75 + (counter - 0.5) / divisions));
689  angle.y = v_forward.x;
690  angle.z = v_forward.y;
691 
692  // first do the spread_to_min effect
693  deviation = angle * spread_to_min;
694  deviation = ((this.sw_shotdir + (right * deviation.y) + (up * deviation.z)));
695  new_min_dist = SW_DISTTOMIN;
696  new_min_end = (this.sw_shotorg + (deviation * new_min_dist));
697  //te_lightning2(NULL, new_min_end, this.sw_shotorg);
698 
699  // then calculate spread_to_max effect
700  deviation = angle * spread_to_max;
701  deviation = ((this.sw_shotdir + (right * deviation.y) + (up * deviation.z)));
702  new_max_dist = vlen(new_min_end - endpos);
703  new_max_end = (new_min_end + (deviation * new_max_dist));
704  //te_lightning2(NULL, new_end, prev_min_end);
705 
706 
707  if(counter == 0)
708  {
709  first_min_end = new_min_end;
710  first_max_end = new_max_end;
711  }
712 
713  if(counter >= 1)
714  {
715  // draw from shot origin to min spread radius
716  R_BeginPolygon("", DRAWFLAG_NORMAL);
717  R_PolygonVertex(prev_min_end, '0 0 0', sw_color, a);
718  R_PolygonVertex(new_min_end, '0 0 0', sw_color, a);
719  R_PolygonVertex(this.sw_shotorg, '0 0 0', sw_color, a);
720  R_EndPolygon();
721 
722  // draw from min spread radius to max spread radius
723  R_BeginPolygon("", DRAWFLAG_NORMAL);
724  R_PolygonVertex(new_min_end, '0 0 0', sw_color, a);
725  R_PolygonVertex(prev_min_end, '0 0 0', sw_color, a);
726  R_PolygonVertex(prev_max_end, '0 0 0', sw_color, a);
727  R_PolygonVertex(new_max_end, '0 0 0', sw_color, a);
728  R_EndPolygon();
729  }
730 
731  prev_min_end = new_min_end;
732  prev_max_end = new_max_end;
733 
734  // last division only
735  if((counter + 1) == divisions)
736  {
737  // draw from shot origin to min spread radius
738  R_BeginPolygon("", DRAWFLAG_NORMAL);
739  R_PolygonVertex(prev_min_end, '0 0 0', sw_color, a);
740  R_PolygonVertex(first_min_end, '0 0 0', sw_color, a);
741  R_PolygonVertex(this.sw_shotorg, '0 0 0', sw_color, a);
742  R_EndPolygon();

747  R_PolygonVertex(prev_min_end, '0 0 0', sw_color, a);
748  R_PolygonVertex(prev_max_end, '0 0 0', sw_color, a);
749  R_PolygonVertex(first_max_end, '0 0 0', sw_color, a);
750  R_EndPolygon();
751  }
